Find a year is leapyear:

#include<stdio.h>
int main(){
    int year;
    printf("year: ");
    scanf("%d",&year);
    if(year%400==0||year%4==0&&year%100!=0){
        printf("leap year");
    }
    else{
        printf("not leap year");
    }
}
