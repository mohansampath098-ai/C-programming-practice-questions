delete an element in an array:

#include<stdio.h>
int main(){
    int n,a[100];
    int pos;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    scanf("%d",&pos);
    pos--;
    for(int i=pos;i<n-1;i++){
        a[i]=a[i+1];
    }
    for(int i=0;i<n-1;i++){
        printf("%d",a[i]);
    }
    return 0;
}
