1. FIND NUMBER OF VOWLES AND THE CONSONENTS IN A STRING.


#include<stdio.h>
int main(){
    char a[100];
    int vow,con;
    fgets(a,sizeof(a),stdin);
    vow=con=0;
    for(int i=0;a[i]!='\0';++i){
        if(a[i]=='a' || a[i]=='e' || a[i]=='i' || a[i]=='o' || a[i]=='u' || a[i]=='A' || a[i]=='E' || a[i]=='I' || a[i]=='O' || a[i]=='U'){
        ++vow;
        }
        else if(a[i]>='a' && a[i]<='z' || a[i]>='A' && a[i]<='Z'){
            ++con;
        }
    }
    printf("%d",vow);
    printf("%d",con);
    
}

-------------->
OUTPUT:
MOHAN
2 3
