Decimal number into the binary number:

#include<stdio.h>
int main(){
    int rem,dec,i=0;
    char bin[32];
    scanf("%d",&dec);
    if(dec==0){
        printf("INVALID NUMBER");
    }
    while(dec>0){
        rem=dec%2;
        bin[i]=rem;
        dec=dec/2;
        i++;
    }
    for(int j=i-1;j>=0;j--){
        printf("%d",bin[j]);
    }
    return 0;
}
