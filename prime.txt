int main(){
    int n;
    int flag;
    flag=0;
    scanf("%d",&n);
    for(int i=2;i<=n;i++){
        if(n%i!=0){
           continue;
        }
        else{
            flag=1;
            printf("prime");
            break;
        }
        if(flag==0){
             printf("not Prime");
        }
    }
}
