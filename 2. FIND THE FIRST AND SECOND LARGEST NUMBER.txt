2. FIND THE FIRST AND SECOND LARGEST NUMBER IN AN ARRAY ?


#include<stdio.h>
int main(){
    int a[5]={10,20,30,40,50};
    int n;
    int max1,max2;
    n=sizeof(a)/sizeof(a[0]);
    if(a[0]>a[1]){
        max1=a[0];
        max2=a[1];
    }
    else{
        max1=a[1];
        max2=a[0];
    }
    for(int i=2;i<n;i++){
        if(a[i]>max1){
            max2=max1;
            max1=a[i];
        }
        else if(a[i]>max2){
            max2=a[i];
        }
    }
    printf(" first largest number:%d\n",max1);
    printf(" second largest number:%d\n",max2);
    return 0;
}



------> THE OUTPUT IS 

first largest number:50
second largest number=40
