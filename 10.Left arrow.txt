        if(i<0){
            l=-i;
        }
        else{
            l=i;
        }
        for(int s=0;s<n-l;s++){                                                     HINT: NO SPACE IN PRINTF

            printf("");
        }
        for(int j=0;j<l+1;j++){

        }
        printf("\n");
        }    
}
