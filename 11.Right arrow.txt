#include<stdio.h>
int main(){
    int n=4,l;
    for(int i=-n;i<n;i++){
        if(i<0){
            l=-i;
        }
        else{
            l=i;
        }                                                                                   HINT: DOUBLE SPACE IN PRINTF

        for(int s=0;s<n-l;s++){
            printf("  ");
        }
        for(int j=0;j<l+1;j++){
            printf("* ");
        }
        printf("\n");
        }    
}
