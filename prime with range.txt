int main(){

    int end;
    int flag;
    scanf("%d %d",&start,&end);
    for(int j=start;j<=end;j++){
        flag=0;
    for(int i=2;i<j;i++){
        if(j%i==0){
            flag=1;
            break;
        }
    }
    if(flag==0){
        printf(" %d",j);
    }
    }
        
}
